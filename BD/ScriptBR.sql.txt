    NOME     VARCHAR(100),
    SALARIO  DOUBLE PRECISION
);

    ISCALCULAIR    SMALLINT DEFAULT 0 NOT NULL,
    ISCALCULAINSS  SMALLINT DEFAULT 0 NOT NULL,
    CPF            VARCHAR(14) NOT NULL
);

ALTER TABLE DEPENDENTE ADD CONSTRAINT PK_DEPENDENTE PRIMARY KEY (ID);
